55  /**
56   * A maven2 mojo that injects elements necessary to run Cactus tests into an
57   * existing EAR file.
58   * 
59   * @version $Id: CactifyEarMojo.java 394252 2008-04-29 04:20:17Z ptahchiev $
60   * @goal cactifyear
61   * @requiresDependencyResolution compile
62   */
63  public class CactifyEarMojo extends AbstractMojo
64  {
65      /**
66       * Cactus war configuration holder.
67       * @parameter
68       */
69      private CactifyWarMojo cactusWar;

72       * The archive that contains the web-app that should be cactified.
73       * @required
74       * @parameter

76      private File srcFile;

79       * The archive that contains the web-app that should be cactified.
80       * @parameter
81       * default-value="${project.build.directory}/${project.artifactId}-cactified.ear"

86       * Indicates whether or not we should add ejb references to local ejbs

130     /**
131      * The "main" method of the mojo.
132      * @throws MojoExecutionException in case an error occurs.
133      * @throws MojoFailureException in case a failure occurs.
134      */
135     public void execute() throws MojoExecutionException, MojoFailureException 
136     {
137         if (cactusWar == null)
138         {
139             cactusWar = createCactusWarConfig();
140         }
141         
142         if (getSrcFile() == null) 
143         {
144             throw new MojoExecutionException("You need to specify [srcFile] "
145                 + "attribute for cactification!");
146         }
147         MavenArchiver archiver = new MavenArchiver();
148         archiver.setArchiver(earArchiver);
149         
150         //Add everything that's in the source EAR to the destination EAR
151         File tempLocation = null;
152         
153         tempLocation = FileUtils.createTempFile("cactus", "explode.tmp.dir",
154                 getProject().getBasedir());
155         tempLocation.mkdirs();
156         tempLocation.deleteOnExit();
157         
158         try 
159         {
160             if (this.srcFile != null)
161             {
162                 AssemblyFileUtils.unpack(this.srcFile, tempLocation,
163                     archiverManager);
164             }
165         } 
166         catch (ArchiveExpansionException e) 
167         {
168             throw new MojoExecutionException("Error extracting the"
169                    + " archive.", e);
170         } 
171         catch (NoSuchArchiverException e) 
172         {
173             throw new MojoExecutionException("Problem reading the "
174                    + "source archive.", e);
175         }
176         
177         try 
178         {
179             earArchiver.addDirectory(tempLocation);
180         } 
181         catch (ArchiverException e1) 
182         {
183             // Cannot add the temp location for some reason.
184             throw new MojoExecutionException("Problem adding the source "
185                 + "files to the dest. archive ", e1);
186         }
187         
188         // cactify the application.xml
189         ApplicationXml appXml = null;
190         try 
191         {
192             appXml = getOriginalApplicationXml();
193         } 
194         catch (JDOMException e) 
195         {
196             throw new MojoExecutionException("Unable to get the "
197                + "original application.xml", e);
198         }
199         
200         File tmpAppXml = cactifyApplicationXml(appXml);
201         
202         try 
203         {
204             earArchiver.addDirectory(tempLocation);
205             earArchiver.setAppxml(tmpAppXml);
206             
207             archiver.setOutputFile(getDestFile());
208             
209             // create the cactus war
210             File cactusWarFile = createCactusWarFile();
211             addFileToEar(cactusWarFile, cactusWar.getFileName());
212             
213             archiver.createArchive(getProject(), getArchive());
214         }
215         catch (ArchiverException aex)
216         {
217             throw new MojoExecutionException("Error while performing the "
218                     + "cactified archive.", aex);
219         } 
220         catch (ManifestException me) 
221         {
222             throw new MojoExecutionException("Error reading the manifest "
223                    + "data in the original file.", me);
224         } 
225         catch (IOException ioe) 
226         {
227             throw new MojoExecutionException("Input/output exception occured ",
228                     ioe);
229         }
230         catch (DependencyResolutionRequiredException de) 
231         {
232             throw new MojoExecutionException("Problem with resolving the"
233                     + " dependencies of the project. ", de);
234         }
235         finally
236         {
237             try 
238             {
239                 if (tempLocation != null)
240                 {
241                     FileUtils.deleteDirectory(tempLocation);
242                 }
243             } 
244             catch (IOException e) 
245             {
246                 throw new MojoExecutionException("Error deleting temporary "
247                        + "folder", e);
248             }
249         }
250     }

252     /**
253      * A helper method to create a temporary file.
254      * @return the cactus.war
255      * @throws MojoExecutionException in case a runtime error occurs.
256      * @throws MojoFailureException in case a mojo failure occurs.
257      */
258     private File createCactusWarFile() 
259         throws MojoExecutionException, MojoFailureException
260     {
261         File tmpCactusWar = FileUtils.createTempFile("cactus", "cactus.war",
262                                                      getProject().getBasedir());
263         tmpCactusWar.deleteOnExit();
264         cactusWar.setDestFile(tmpCactusWar);
265         cactusWar.setWarArchiver(this.warArchiver);
266         cactusWar.setProject(this.project);
267         
268         if (addEjbReferences)
269         {
270             try 
271             {
272                 addEjbReferencesToWar(tmpCactusWar);
273             } 
274             catch (JDOMException e) 
275             {
276                 throw new MojoExecutionException(
277                         "Unable to add ejb-references", e);
278             }
279         }
280         
281         cactusWar.execute();
282         
283         return tmpCactusWar;
284     }

286     /**
287      * 
288      * @param theAppXml the application.xml to cactify
289      * @return the cactified application.xml
290      * @throws MojoExecutionException in case a problem occurs
291      */
292     private File cactifyApplicationXml(ApplicationXml theAppXml) 
293                                             throws MojoExecutionException
294     {
295         theAppXml.addWebModule(cactusWar.getFileName(), cactusWar.getContext());
296         // serialize the cactified app xml
297         //FileUtils fileUtils = FileUtils.newFileUtils();
298         File tmpAppXml = FileUtils.createTempFile("cactus", "application.xml",
299                                                   getProject().getBasedir());
300         tmpAppXml.deleteOnExit();
301         try
302         {
303             ApplicationXmlIo.writeApplicationXml(theAppXml, 
304                                                  tmpAppXml, 
305                                                  null, true);
306         }
307         catch (IOException ioe)
308         {
309             throw new MojoExecutionException(
310                 "Could not write temporary deployment descriptor", ioe);
311         }
312         return tmpAppXml;
313     }

315     /**
316      * 
317      * @return the application.xml from the source ear
318      * @throws JDOMException in case a JDOM exception is thrown
319      * @throws MojoExecutionException in case a problem occurs
320      */
321     private ApplicationXml getOriginalApplicationXml() 
322                             throws JDOMException, MojoExecutionException
323     {
324         ApplicationXml appXml = null;
325         try
326         {
327             EarArchive ear = new DefaultEarArchive(
328                 new FileInputStream(this.srcFile));
329             
330             appXml = ear.getApplicationXml();
331             if (appXml == null)
332             {
333                 throw new MojoExecutionException("The EAR source file does not "
334                     + "contain a META-INF/application.xml " 
335                     + "deployment descriptor");
336             }
337         }
338         catch (IOException e)
339         {
340             throw new MojoExecutionException("Failed to open EAR", e);
341         }
342         
343         return appXml;
344     }

346     /**
347      * 
348      * @param theFile the file to add
349      * @param theFullPath the path within the ear
350      * @throws ArchiverException in case the adding is impossible.
351      */
352     private void addFileToEar(File theFile, String theFullPath) 
353                                                     throws ArchiverException
354     {
355         earArchiver.addFile(theFile, theFullPath);
356     }

358     /**
359      * Add ejb references.
360      * 
361      * @param theWar temporary cactus war
362      * @throws JDOMException in case a parse exception is thrown 
363      * @throws MojoExecutionException in case any other error occurs.
364      */
365     private void addEjbReferencesToWar(File theWar) 
366                     throws JDOMException, MojoExecutionException 
367     {
368         try
369         {
370             EarArchive ear = new DefaultEarArchive(
371                 new FileInputStream(srcFile));
372             ApplicationXml appXml = ear.getApplicationXml();
373             Iterator ejbModules = appXml.getEjbModules();
374             while (ejbModules.hasNext())
375             {
376                 String module = (String) ejbModules.next();
377                 EjbArchive ejbArchive = ear.getEjbModule(module);
378                 EjbJarXml descr = ejbArchive.getEjbJarXml();
379                 Iterator vendorDescrIterator = descr.getVendorDescriptors();
380                 if (vendorDescrIterator == null 
381                     || !vendorDescrIterator.hasNext())
382                 {
383                     throw new MojoExecutionException("Failed to find vendor " 
384                                              + "deployment descriptor " 
385                                              + "for ejb jar " + module);
386                 }
387                 
388                 Iterator ejbs = descr.getSessionEjbs();
389                 while (ejbs.hasNext())
390                 {
391                     Session ejb = (Session) ejbs.next();
392                     String name = ejb.getName();
393                     String local = ejb.getLocal();
394                     String localHome = ejb.getLocalHome();
395                     if (local != null)
396                     {
397                         getLog().debug("Adding ejb-ref for local session ejb "
398                             + ejb.getName());
399                         EjbRef ref = new EjbRef();
400                         ref.setType("Session");
401                         ref.setEjbName(name);
402                         ref.setName("ejb/" + name);
403                         ref.setEjbInterface(local);
404                         ref.setEjbHomeInterface(localHome);
405                         ref.setLocal(true);
406                         
407                         cactusWar.addConfiguredEjbref(ref);
408                     }
409                 }
410                 ejbs = descr.getEntityEjbs();
411                 while (ejbs.hasNext())
412                 {
413                     Entity ejb = (Entity) ejbs.next();
414                     String name = ejb.getName();
415                     String local = ejb.getLocal();
416                     String localHome = ejb.getLocalHome();
417                     if (local != null)
418                     {
419                         getLog().debug("Adding ejb-ref for local entity ejb "
420                             + ejb.getName());
421                         EjbRef ref = new EjbRef();
422                         ref.setType("Entity");
423                         ref.setEjbName(name);
424                         ref.setName("ejb/" + name);
425                         ref.setEjbInterface(local);
426                         ref.setEjbHomeInterface(localHome);
427                         ref.setLocal(true);
428                         
429                         cactusWar.addConfiguredEjbref(ref);
430                     }
431                 }
432             }
433         }
434         catch (IOException e)
435         {
436             throw new MojoExecutionException("Could not merge deployment " 
437                                      + "descriptors", e);
438         }
439     }

441     /**
442      * Initialize cactusWar with some default values.
443      *
444      * @return the CactusWar configuration
445      */
446     private CactifyWarMojo createCactusWarConfig()
447     {
448         CactifyWarMojo cactusWarConfig = new CactifyWarMojo();
449         Version version = new Version();
450         version.setValue("2.3");
451         cactusWarConfig.setVersion(version);
452         cactusWarConfig.setContext("/cactus");
453         cactusWarConfig.setWarArchiver(warArchiver);
454         cactusWarConfig.setProject(getProject());
455         
456         return cactusWarConfig;
457     }
