44  /**
45   * Provides a launcher to start Cactus tests. This is done by extending
46   * the JUnit Plugin launch configuration and adding Cactus specific VM
47   * configuration (Cactus jars, VM parameters) and by registering this
48   * class as an 
49   * <code>"org.eclipse.debug.core.launchConfigurationTypes"</code> Eclipse
50   * extension point.
51   * 
52   * @version $Id: JettyCactusLaunchConfiguration.java 238816 2004-02-29 16:36:46Z vmassol $
53   */
54  public class JettyCactusLaunchConfiguration
55      extends CactusLaunchConfiguration
56  {
57      /**
58       * Id under which the Cactus launch configuration has been registered. 
59       */
60      public static final String ID_CACTUS_APPLICATION_JETTY =
61          "org.apache.cactus.eclipse.runner.launchconfig.jetty";
62  
63      /**
64       * Path to the Jetty library directory in the Cactus plugin
65       * structure
66       */
67      private static final String JETTY_LIBRARY_PATH = "./lib/";
68  
69      /**
70       * @return an array of classpaths needed for Cactus
71       * @throws CoreException when an error occurs while
72       * trying to build the classpath
73       */
74      protected IClasspathEntry[] getCactusClasspath() throws CoreException
75      {
76          IClasspathEntry[] cactusClasspath = super.getCactusClasspath();
77          try {
78          	cactusClasspath =
79                  LibraryHelper.concatenateEntries(
80                      cactusClasspath,
81                      getJettyClasspath());
82          } catch (MalformedURLException me) {
83          	throw CactusPlugin.createCoreException(
84                      "CactusLaunch.message.prepare.error.plugin.file",
85                      " : " + me.toString(),
86                      null);
87          }
88          return cactusClasspath;
89      }
90  
91      /**
92       * @return an array of Jar paths needed for Cactus
93       * @throws CoreException if a Jar cannot be found
94       * @throws MalformedURLException 
95       */
96      private IClasspathEntry[] getJettyClasspath() throws CoreException, MalformedURLException
97      {
98          CactusPlugin thePlugin = CactusPlugin.getDefault();
99         
100         IPath libURL = LibraryHelper.getLibPath();
101         
102         if (libURL == null)
103         {
104             throw CactusPlugin.createCoreException(
105                 "CactusLaunch.message.prepare.error.plugin.file",
106                 " : " + libURL.toString(),
107                 null);
108         }
109         File libDir = new File(libURL.toString());
110         IClasspathEntry[] jettyJarPaths = getJarPaths(libDir);
111         IClasspathEntry[] apacheJarPaths = new IClasspathEntry[0];
112         Plugin tomcatPlugin = Platform.getPlugin("org.eclipse.tomcat");
113         if (tomcatPlugin != null)
114         {
115             IPluginDescriptor descriptor = tomcatPlugin.getDescriptor();
116             apacheJarPaths = getLibrariesPaths(descriptor, "org.apache.jasper");
117         }
118         return LibraryHelper.concatenateEntries(jettyJarPaths, apacheJarPaths);
119     }
120 
121     /**
122      * @param theJavaProject the Java project to get the arguments for
123      * @return an array of the specific Cactus VM arguments
124      */
125     protected String getCactusVMArgs(IJavaProject theJavaProject)
126     {
127         String cactusVMArgs = super.getCactusVMArgs(theJavaProject);
128         String jettyResourcePath =
129             CactusPreferences.getTempDir()
130                 + File.separator
131                 + JettyContainerManager.getJettyWebappPath();
132         cactusVMArgs += "-Dcactus.jetty.resourceDir="
133             + jettyResourcePath
134             + VM_ARG_SEPARATOR;
135         IPath projectLocation = theJavaProject.getProject().getLocation();
136         File jettyXML =
137             projectLocation.append(CactusPreferences.getJettyXML()).toFile();
138         if (jettyXML.exists())
139         {
140             cactusVMArgs += "-Dcactus.jetty.config="
141                 + jettyXML.getAbsolutePath()
142                 + VM_ARG_SEPARATOR;
143         }
144         cactusVMArgs += "-Dcactus.initializer="
145             + "org.apache.cactus.extension.jetty.JettyTestSetup";
146         return cactusVMArgs;
147     }
148 
149     /**
150      * @param theDirectory the directory to list jars from
151      * @return the array of jar paths in the given directory
152      */
153     private IClasspathEntry[] getJarPaths(File theDirectory)
154     {
155         File[] jars = theDirectory.listFiles(new JarFilenameFilter());
156         IClasspathEntry[] jarPaths = new IClasspathEntry[jars.length];
157         for (int i = 0; i < jarPaths.length; i++)
158         {
159             jarPaths[i] =
160                 LibraryHelper.getIClasspathEntry(jars[i].getAbsolutePath());
161         }
162         return jarPaths;
163     }
164 
165     /**
166      * @param theDescriptor the plug-in descriptor to get libraries from
167      * @param thePackagePrefix package prefix used to filter libraries 
168      * @return an array of jar paths exposed by the plug-in
169      * @throws MalformedURLException 
170      */
171     private IClasspathEntry[] getLibrariesPaths(
172         IPluginDescriptor theDescriptor, String thePackagePrefix) throws MalformedURLException
173     {
174         Vector result = new Vector();
175         URL root = new URL(("file://"+JavaCore.getClasspathVariable("ECLIPSE_HOME").toFile().toURL().getPath()+theDescriptor.getInstallURL().getPath()).replaceAll("plugin","plugins"));
176         ILibrary[] libraries = theDescriptor.getRuntimeLibraries();
177         
178         for (int i = 0; i < libraries.length; i++)
179         {
180         	
181             ILibrary currentLib = libraries[i];
182             //CactusPlugin.log("Current library "+currentLib.getPath().toString());
183             if (thePackagePrefix == null
184                 || isContained(thePackagePrefix, currentLib))
185             {
186                 try
187                 {
188                     URL url = new URL(root, currentLib.getPath().toString());
189                     result.add(
190                         LibraryHelper.getIClasspathEntry(
191                             Platform.asLocalURL(url).getPath()));
192                 }
193                 catch (IOException e)
194                 {
195                     // if the URL is not valid we don't add it
196                     CactusPlugin.log(e);
197                     continue;
198                 }
199             }
200         }
201         return (IClasspathEntry[]) result.toArray(
202             new IClasspathEntry[result.size()]);
203     }
204 
205     /**
206      * @param thePackagePrefix prefix which presence is to be tested
207      * @param theCurrentLib the library in which the prefix will be searched 
208      * @return true if the library declares the given package prefix
209      */
210     private boolean isContained(String thePackagePrefix, ILibrary theCurrentLib)
211     {
212     	String[] prefixes = theCurrentLib.getPackagePrefixes();
213         if(prefixes==null || prefixes.length==0) {
214         	//CactusPlugin.log("lib: "+theCurrentLib.getPath().toString()+" is not");
215         	return false;
216         }
217         for (int i = 0; i < prefixes.length; i++)
218         {
219             if (prefixes[i].equals(thePackagePrefix))
220             {
221                 return true;
222             }
223         }
224         return false;
225     }
